use ui::Ui;

/// Event callback type
pub type EventCallback<ID> = Fn(&Ui<ID>, &ID, &Event, &EventArgs) -> ();

/**
    Events name definition
*/
#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub enum Event {
    Clicked,
    KeyDown,
    KeyUp,
    Char,
    MenuOpen,
    MenuClose,
    Closed,
    Destroyed,
}

/**
    Events arguments definition. If an event do not have arguments, EventArgs::None is passed.
*/
pub enum EventArgs {
    Key(u32),
    Char(char),
    None
}
